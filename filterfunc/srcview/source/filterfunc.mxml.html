<?xml version="1.0" encoding="utf-8"?>
<mx:Application xmlns:mx="http://www.adobe.com/2006/mxml" creationComplete="dp.filterFunction=filterFunc" layout="absolute" width="300" height="350" backgroundColor="#ffffff" backgroundGradientColors="[#ffffff,#ffffff]" viewSourceURL="srcview/index.html">
    
    <mx:Script>
        <![CDATA[
            import mx.collections.SortField;
            import mx.collections.Sort;
            import mx.collections.ArrayCollection;
            [Bindable]private var dp:ArrayCollection=new ArrayCollection(["Israel","Afghanistan","Argentina","Austria","Bolivia","Brazil","Canada",
                "China","Denmark","Egypt","Finland","France","Germany","Guatemala","Honduras","Iceland","Italy","Japan","Latvia","Mexico"]);
            
            private function filterFunc(item:Object):Boolean
            {
                return item.toString().toLowerCase().indexOf(filterText.text.toLowerCase())!=-1;
            }
            private function sortList():void{
                var s:Sort=new Sort();
                s.fields=[new SortField(null,false,true)];
                dp.sort=s;
                dp.refresh();            
            }
        ]]>
    </mx:Script>
    
    <mx:VBox width="100%" height="100%" paddingBottom="2" paddingTop="5" paddingLeft="5" paddingRight="5">
        <mx:HBox width="100%" height="30" verticalAlign="middle">
            <mx:Label text="Filter:"/>
            <mx:TextInput id="filterText" width="100%" change="dp.refresh()"/>
            <mx:Button label="sort" click="sortList()"/>
        </mx:HBox>
        <mx:List width="100%" height="100%" dataProvider="{dp}"/>
    </mx:VBox>
</mx:Application>

<s:Application xmlns:fx="http://ns.adobe.com/mxml/2009" 
               xmlns:s="library://ns.adobe.com/flex/spark" 
               xmlns:com="com.*" 
               xmlns:mx="library://ns.adobe.com/flex/mx" width="720" height="344" backgroundColor="#ffffff"
               creationComplete="srv.send()" viewSourceURL="srcview/index.html">
    <fx:Declarations>
        <s:HTTPService id="srv" url="data/Pictures.xml" result="onResult(event)"/>
    </fx:Declarations>
    
    <fx:Script>
        <![CDATA[
            import mx.collections.ArrayCollection;
            import mx.rpc.events.ResultEvent;
            import flash.events.MouseEvent;
            
            [Bindable]private var dp:ArrayCollection;
            
            protected function onResult(event:ResultEvent):void
            {
                dp=event.result.pictures.picture;
            }
                    
            
            protected function GoToItem(event:MouseEvent):void
            {
                var newPosition:int=hl.width/dp.length*ns2.value;
                asc.ScrollToPosition(newPosition);
            }
            
        ]]>
    </fx:Script>    
    
    <s:layout>
        <s:VerticalLayout paddingLeft="10" paddingTop="10" paddingRight="10" gap="16"/>
    </s:layout>
    
    <s:Label fontSize="16" fontWeight="bold" text="AnimatedScrollCanvas Demo"/>
    <s:HGroup verticalAlign="middle">
        <s:Label text="Max scroll speed:"/>
        <s:NumericStepper id="ns" minimum="0" maximum="100" value="15"/>
    </s:HGroup>
    
    <com:AnimatedScrollCanvas id="asc" Speed="{ns.value}" borderStyle="solid" width="100%" height="220">
        <mx:HorizontalList id="hl" borderStyle="none" liveScrolling="true" dataProvider="{dp}" width="{dp.length*260}" height="220" horizontalScrollPolicy="off" verticalScrollPolicy="off" itemRenderer="renderers.imageRenderer"/>    
    </com:AnimatedScrollCanvas>

    <s:HGroup verticalAlign="middle">
        <s:Button label="Scroll to position" click="asc.ScrollToPosition(ns1.value)"/>
        <s:NumericStepper id="ns1" minimum="0" maximum="{hl.width}" stepSize="50"/>
        <s:Label text="Max Position: {hl.width}"/>
        <mx:Spacer width="140"/>
        <s:Button label="Go to item" click="GoToItem(event)"/>
        <s:NumericStepper id="ns2" minimum="0" maximum="{dp.length-1}"/>    
        <s:Label text="Total Items: {dp.length}"/>         
    </s:HGroup>
    
</s:Application>

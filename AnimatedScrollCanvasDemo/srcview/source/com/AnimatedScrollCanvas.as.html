package com
{
    import flash.events.Event;
    import flash.events.MouseEvent;
    import mx.containers.Canvas;
    
    public class AnimatedScrollCanvas extends Canvas
    {
        private var _speed:int=15;
        private var animate:Boolean=false;
        private var scrollingToPosition:Boolean=false;
        private var targetScrollPosition:int=0;

        public function AnimatedScrollCanvas()
        {
            super();
            this.horizontalScrollPolicy="off";
            this.verticalScrollPolicy="off";
            this.clipContent=true;
            this.addEventListener(Event.ENTER_FRAME, onEnterFrame);
            this.addEventListener(MouseEvent.ROLL_OVER, onMouseInCanvas);
            this.addEventListener(MouseEvent.ROLL_OUT, onMouseOutCanvas);
        }

        private function onMouseInCanvas(e:MouseEvent):void{
            animate=true;
        }
        private function onMouseOutCanvas(e:MouseEvent):void{
            animate=false;
        }
        private function onEnterFrame(e:Event):void{
            if (animate && !scrollingToPosition){
                var currentSpeed:int;    
                if (this.mouseX<this.width/2){
                    currentSpeed=(mouseX-this.width/2)/(this.width/2)*_speed;
                    if (this.horizontalScrollPosition>0)
                        this.horizontalScrollPosition+=currentSpeed;                        
                }else{
                    currentSpeed=(this.width/2-mouseX)/(this.width/2)*_speed;
                    if (this.horizontalScrollPosition<=this.maxHorizontalScrollPosition)
                        this.horizontalScrollPosition-=currentSpeed;                        
                }            
            }
            if (scrollingToPosition){
                var fastScrollSpeed:int=Math.abs(targetScrollPosition-this.horizontalScrollPosition)/stage.frameRate+_speed;
                if (fastScrollSpeed<=_speed){
                    this.horizontalScrollPosition=targetScrollPosition;
                    scrollingToPosition=false;
                    return;
                }
                if (this.horizontalScrollPosition>targetScrollPosition){
                    this.horizontalScrollPosition-=fastScrollSpeed;
                }else{
                    this.horizontalScrollPosition+=fastScrollSpeed;
                }                
            }
        }
        
// ***************** API Functions ************************************************
        
        public function ScrollToPosition(position:int):void{
            position>this.maxHorizontalScrollPosition?targetScrollPosition=this.maxHorizontalScrollPosition:targetScrollPosition=position;
            scrollingToPosition=true;
        }        
        
// ***************** Getters / Setters *********************************************
        
        public function get Speed():int
        {
            return _speed;
        }
        
        public function set Speed(value:int):void
        {
            _speed = value;
        }        
        
    }
    
// **********************************************************************************    
}

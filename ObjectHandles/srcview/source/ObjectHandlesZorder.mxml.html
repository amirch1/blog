<?xml version="1.0" encoding="utf-8"?>
<s:Application xmlns:fx="http://ns.adobe.com/mxml/2009" 
               xmlns:s="library://ns.adobe.com/flex/spark" 
               xmlns:mx1="library://ns.adobe.com/flex/mx"
               xmlns:objecthandles="com.roguedevelopment.objecthandles.*"
               xmlns:example="com.roguedevelopment.objecthandles.example.*"
               xmlns:mx="library://ns.adobe.com/flex/mx" width="420" height="500" backgroundColor="#E8E8E8"
               click="application_clickHandler(event)" viewSourceURL="srcview/index.html">
    
    <fx:Script>
        <![CDATA[
            import com.roguedevelopment.objecthandles.*;
            import com.roguedevelopment.objecthandles.HandleDescription;
            import com.roguedevelopment.objecthandles.HandleRoles;
            import com.roguedevelopment.objecthandles.decorators.AlignmentDecorator;
            import com.roguedevelopment.objecthandles.decorators.DecoratorManager;
            import com.roguedevelopment.objecthandles.example.*;
            
            import mx.controls.Image;
            import mx.core.FlexLoader;
            import mx.core.IVisualElement;
                    
            protected var objectHandles:ObjectHandles;
            protected var decoratorManager:DecoratorManager;

            
            override protected function initializationComplete() : void
            {        
                objectHandles = new ObjectHandles( mainGroup, null, new Flex4HandleFactory(), new Flex4ChildManager());
                
                for (var i:int=1; i<6; i++){
                    var flexModel:SimpleDataModel=new SimpleDataModel();
                    flexModel.x=50*i;
                    flexModel.y=50*i;
                    flexModel.width=120;
                    flexModel.height=80;
                    
                    var flexShape:SimpleFlex4Shape=new SimpleFlex4Shape();
                    flexShape.model=flexModel;
                    var img:Image=new Image();
                    img.percentHeight=img.percentWidth=100;
                    img.maintainAspectRatio=false;
                    img.load("assets/pic"+i+".jpg");
                    flexShape.addElement(img);
                    mainGroup.addElement(flexShape);
                    
                    objectHandles.registerComponent( flexModel, flexShape );
                }

                super.initializationComplete();
                
                decoratorManager = new DecoratorManager( objectHandles, drawingLayer );
                decoratorManager.addDecorator( new AlignmentDecorator() );
            }
            
            protected function application_clickHandler(event:MouseEvent):void
            {
                if(event.target is SimpleFlex4Shape) return;
                if(event.target is VisualElementHandle) return;
                if(event.target is FlexLoader) return;
                if(event.target is Button) return;
                objectHandles.selectionManager.clearSelection();
            }
            
            protected function getSelectedObject():IVisualElement{
                // getting the selected IVisualElement by the selected model in the objectHandles selectionManager
                var ve:IVisualElement;
                if (objectHandles.selectionManager.currentlySelected[0]!=undefined){
                    for (var i:int=0; i<mainGroup.numChildren; i++){
                        if (mainGroup.getElementAt(i) is SimpleFlex4Shape){
                            if((mainGroup.getElementAt(i) as SimpleFlex4Shape).model==objectHandles.selectionManager.currentlySelected[0]){
                                ve=mainGroup.getElementAt(i) as IVisualElement;
                                break;
                            }
                        }
                    }
                }
                return ve;
            }

            protected function MoveBack(event:MouseEvent):void
            {
                var ve:IVisualElement=getSelectedObject();
                if (ve){
                    var currentIndex:int=mainGroup.getElementIndex(ve);
                    if (currentIndex>0)
                        currentIndex--;
                    mainGroup.setElementIndex(ve,currentIndex);
                }
            }


            protected function BringForward(event:MouseEvent):void
            {
                var ve:IVisualElement=getSelectedObject();
                if (ve){
                    var currentIndex:int=mainGroup.getElementIndex(ve);
                    if (currentIndex<mainGroup.numChildren-1)
                        currentIndex++;
                    mainGroup.setElementIndex(ve,currentIndex);
                    // need to refresh the selection handles to make sure its above the image
                    objectHandles.selectionManager.setSelected(objectHandles.selectionManager.currentlySelected[0]);
                }
            }


            protected function SendToBack(event:MouseEvent):void
            {
                var ve:IVisualElement=getSelectedObject();
                if (ve){
                    var currentIndex:int=mainGroup.getElementIndex(ve);
                    if (currentIndex>0)
                        mainGroup.setElementIndex(ve,0);
                }
            }


            protected function BringToFront(event:MouseEvent):void
            {
                var ve:IVisualElement=getSelectedObject();
                if (ve){
                    var currentIndex:int=mainGroup.getElementIndex(ve);
                    if (currentIndex<mainGroup.numChildren-1)
                        mainGroup.setElementIndex(ve,mainGroup.numChildren-1);
                    // need to refresh the selection handles to make sure its above the image
                    objectHandles.selectionManager.setSelected(objectHandles.selectionManager.currentlySelected[0]);
                }
            }

        ]]>
    </fx:Script>    
    
    <s:controlBarContent>
        <s:Button label="Move Back" click="MoveBack(event)"/>
        <s:Button label="Bring Forward" click="BringForward(event)"/>
        <s:Button label="Send To Back" click="SendToBack(event)"/>
        <s:Button label="Bring To Front" click="BringToFront(event)"/>
    </s:controlBarContent>
    
    <s:SpriteVisualElement width="100%" height="100%" id="drawingLayer" />    
    <s:Group id="mainGroup" width="100%" height="100%"/>

</s:Application>
